1. 页面的水印: css和js, 为了防止用户修改: 仅用鼠标右键, 控制台打开的自动debugger
```css
body {
  background-image: url('/path/to/watermark.png');
  background-repeat: repeat-y; /* 或者repeat-x, no-repeat 等 */
  background-position: center; /* 或者 top right 等 */
  background-attachment: fixed; /* 使水印固定在页面上 */
}
```

```javascript
document.addEventListener('DOMContentLoaded', function() {
  var watermark = document.createElement('div');
  watermark.style.position = 'absolute';
  watermark.style.opacity = '0.1';
  watermark.style.fontSize = '3em';
  watermark.style.color = '#999';
  watermark.style.zIndex = '1000';
  watermark.style.left = '50%';
  watermark.style.top = '50%';
  watermark.style.transform = 'translate(-50%, -50%)';
  watermark.textContent = '公司名称';
  document.body.appendChild(watermark);
});
```
2. 为图片加水印

```html
<!DOCTYPE html>
<html>
<head>
    <title>Add Watermark to Image</title>
</head>
<body>
    <input type="file" id="imageInput" />
    <button onclick="addWatermark()">Add Watermark</button>
    <canvas id="canvas" style="display:none;"></canvas>
    <img id="resultImage" crossorigin="anonymous" src="" alt="Watermarked Image" />

    <script>
        document.getElementById('imageInput').addEventListener('change', function(e) {
            var file = e.target.files[0];
            if (!file) return;

            var reader = new FileReader();
            reader.onload = function(event) {
                var img = new Image();
                img.onload = function() {
                    var canvas = document.getElementById('canvas');
                    canvas.width = img.width;
                    canvas.height = img.height;
                    var ctx = canvas.getContext('2d');

                    // 绘制原始图片
                    ctx.drawImage(img, 0, 0, img.width, img.height);

                    // 设置水印的样式
                    ctx.font = 'bold 24px Arial';
                    ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
                    ctx.textAlign = 'center';
                    ctx.fillText('Watermark Text', img.width / 2, img.height - 20);

                    // 展示带有水印的图片
                    document.getElementById('resultImage').src = canvas.toDataURL();
                };
                img.src = event.target.result;
            };
            reader.readAsDataURL(file);
        });

        function addWatermark() {
            var canvas = document.getElementById('canvas');
            // 可以在这里添加更多逻辑来处理水印
            alert('Watermark added!');
        }
    </script>
</body>
</html>
```
