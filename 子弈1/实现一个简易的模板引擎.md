实现一个简易的模板引擎
const template = '嗨，{{ info.name.value }}您好，今天是星期 {{ day.value }}';

const data = {
  info: {
    name: {
      value: '张三'
    }
  },
  day: {
    value: '三'
  }
};

render(template, data); // 嗨，张三您好，今天是星期三


```javascript
function get(objectLike, path, defaultValue) {
  const pathArray = path.split('.');
  let cloneObjectLike = JSON.parse(JSON.stringify(objectLike));
  while(pathArray.length) {
    const key = pathArray.shift();
    cloneObjectLike = (cloneObjectLike||{})[key] || defaultValue;
  }
  return cloneObjectLike;
}


function render(template, data) {
  return template.replace(/\{\{([^\{\}]+)\}\}/g, (value) => {
    return get(data, value.replace(/[\{\{\}\}\s]/g,''))
  })
}
```
